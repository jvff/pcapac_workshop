int main(int argc, char* argv[]) {
    cl_platform_id platforms[10];
    int numPlatforms;

    clGetPlatformIDs(10, platforms, &numPlatforms);

    for (int platformIndex = 0; platformIndex < numPlatforms; ++platformIndex) {
        char* platformName;
        char* platformVendor;

        clGetPlatformInfo(platforms[platformIndex],
                CL_PLATFORM_NAME,
                sizeof(platformName),
                &platformName,
                NULL);

        clGetPlatformInfo(platforms[platformIndex],
                CL_PLATFORM_VENDOR,
                sizeof(platformVendor),
                &platformVendor,
                NULL);

        printf("Platform: %s\n", platformName);
        printf("Vendor: %s\n", platformVendor);
    }

    return 0;
}
